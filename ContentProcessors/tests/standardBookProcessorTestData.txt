                                        const commentContainer = document.getElementById('novel-comment');
                                        const loadMoreButton = document.getElementById('load-more-comments');

                                            if (response.ok) {
                                                // Lấy nội dung HTML từ API
                                                const html = await response.text();
                                                const commentHeaderCount = (html.match(/main-comment /g) || []).length;
                                
                                                // Gắn nội dung HTML vào `#novel-comment`
                                                commentContainer.innerHTML = html;
                                                if (html.trim() == '' || commentHeaderCount < 10) {
                                                    loadMoreButton.style.display = "none";
                                                } else {
                                                    loadMoreButton.style.display = "block";
                                                }
                                                // Hiển thị phần tử
                                                commentContainer.style.display = "block";
                                            } else {
                                                console.error('Failed to load comments:', response.status);
                                                commentContainer.innerHTML = `<p>Error loading comments. Please try again later.</p>`;
                                                commentContainer.style.display = "block";
                                            }
                                        } catch (error) {
                                            console.error('Error fetching comments:', error);
                                            commentContainer.innerHTML = `<p>Error loading comments. Please try again later.</p>`;
                                            commentContainer.style.display = "block";
                                        } finally {
                                            loadingMessage.style.display = 'none';

                                        try {
                                            // Gọi API để lấy dữ liệu thêm
                                            const response = await fetch(`/ajax/comments?skip=${skip}&novel_id=titan-king-ascension-of-the-giant&sort=${sortOrder}`);
                                
                                            if (response.ok) {
